# Usage
usage: CoActP.py [-h] -p1 P1 -p2 P2 -r R [-d D] [-m M] [-g G] -t T [-o O]

# Optional arguments:
  -h, --help  show this help message and exit
  -p1 P1      Peaks file 1
  -p2 P2      Peaks file 2
  -r R        Reference gene annotations in bed format
  -d D        Peak pattern span distance
  -m M        No of peaks per pattern
  -g G        DEG gene list
  -t T        No of threads
  -o O        Output file
